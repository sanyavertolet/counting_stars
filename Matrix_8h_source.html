678 auto operator-(const Matrix<TValueLeft> &lhs, const Matrix<TValueRight> &rhs) {
679  if (lhs.capacity != rhs.capacity) {
680  throw IllegalCapacityException(lhs.capacity, rhs.capacity);
681  }
682  Matrix<decltype(TValueLeft() + TValueRight())> result(lhs.capacity, lhs.mass_transform - rhs.mass_transform);
683  for (auto &[key, value]: lhs.data) {
684  result.data[key] = value;
685  }

689  } else {
690  result.data[key] = -value;
691  }
692  }
693  result.delete_zero_elements();

697 template<typename TValueLeft, typename TValueRight>
698 auto operator*(Matrix<TValueLeft> lhs, Matrix<TValueRight> rhs) {
699  if (lhs.capacity.get_j() != rhs.capacity.get_i()) {
700  throw IllegalCapacityException(lhs.capacity, rhs.capacity);
701  }
702  lhs.apply_mass_transform();
703  rhs.apply_mass_transform();
704  Matrix<decltype(TValueLeft() + TValueRight())> result({lhs.capacity.get_i(), rhs.capacity.get_j() });
705  for (StringInt i = 0; i < lhs.capacity.get_i(); ++i) {
706  for (StringInt j = 0; j < rhs.capacity.get_j(); ++j) {
