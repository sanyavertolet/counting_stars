676 auto operator-(const Matrix<TValueLeft> &lhs, const Matrix<TValueRight> &rhs) {
677  if (lhs.dim != rhs.dim) {
678  throw IllegalDimException(lhs.dim, rhs.dim);
679  }
680  Matrix<decltype(TValueLeft() + TValueRight())> result(lhs.dim, lhs.mass_transform - rhs.mass_transform);
681  for (auto &[key, value]: lhs.data) {
682  result.data[key] = value;
683  }
684  for (auto &[key, value]: rhs.data) {
685  if (result.data.find(key) != result.data.end()) {
686  result.data[key] -= value;
687  } else {
688  result.data[key] = -value;
689  }
690  }
691  result.delete_zero_elements();
692  return result;

695 template<typename TValueLeft, typename TValueRight>
696 auto operator*(Matrix<TValueLeft> lhs, Matrix<TValueRight> rhs) {
697  if (lhs.dim.get_j() != rhs.dim.get_i()) {
698  throw IllegalDimException(lhs.dim, rhs.dim);
699  }
700  lhs.apply_mass_transform();
701  rhs.apply_mass_transform();
702  Matrix<decltype(TValueLeft() + TValueRight())> result({ lhs.dim.get_i(), rhs.dim.get_j() });
703  for (StringInt i = 0; i < lhs.dim.get_i(); ++i) {
704  for (StringInt j = 0; j < rhs.dim.get_j(); ++j) {
